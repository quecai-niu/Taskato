**Files:**

- [ ] **Step 1: 替换 SearchTasksAsync 方法**

将现有的 `SearchTasksAsync`（第 117-137 行）替换为支持筛选排序的版本：

```csharp
/// <summary>
/// 按条件查询历史任务（用于历史查询窗体）
/// 支持状态筛选 + 多条件排序，结果在内存中排序（历史数据量小，不影响性能）
/// </summary>
/// <param name="statusFilter">0=全部, 1=未完成, 2=已完成</param>
/// <param name="sortField">"CreatedAt" / "CompletedAt" / "LastModifiedAt"</param>
/// <param name="sortDesc">true=降序</param>
/// <param name="priorityFirst">true=先按优先级降序排</param>
public async Task<List<TaskItem>> SearchTasksAsync(
    DateTime startDate, DateTime endDate, string keyword = "",
    int statusFilter = 0, string sortField = "CreatedAt",
    bool sortDesc = true, bool priorityFirst = false)
{
    var endDateTime = endDate.Date.AddDays(1);

    var query = _db.Table<TaskItem>()
        .Where(t => t.CreatedAt >= startDate.Date && t.CreatedAt < endDateTime);

    var results = await query.ToListAsync();

    // 关键词筛选（内存中）
    if (!string.IsNullOrWhiteSpace(keyword))
    {
        results = results
            .Where(t => t.Title.Contains(keyword, StringComparison.OrdinalIgnoreCase))
            .ToList();
    }

    // 状态筛选（内存中）
    if (statusFilter == 1)
        results = results.Where(t => !t.IsCompleted).ToList();
    else if (statusFilter == 2)
        results = results.Where(t => t.IsCompleted).ToList();

    // 排序（内存中 — 历史查询数据量通常很小）
    results = ApplySort(results, sortField, sortDesc, priorityFirst);

    return results;
}

/// <summary>对任务列表应用多条件排序</summary>
private static List<TaskItem> ApplySort(
    List<TaskItem> tasks, string sortField, bool sortDesc, bool priorityFirst)
{
    IOrderedEnumerable<TaskItem> ordered;

    if (priorityFirst)
    {
        ordered = sortDesc
            ? tasks.OrderByDescending(t => t.Priority)
            : tasks.OrderBy(t => t.Priority);

        ordered = ThenByField(ordered, sortField, sortDesc);
    }
    else
    {
        ordered = OrderByField(tasks, sortField, sortDesc);
    }

    return ordered.ToList();
}

private static IOrderedEnumerable<TaskItem> OrderByField(
    IEnumerable<TaskItem> source, string field, bool desc)
{
    return field switch
    {
        "CompletedAt" => desc
            ? source.OrderByDescending(t => t.CompletedAt ?? DateTime.MinValue)
            : source.OrderBy(t => t.CompletedAt ?? DateTime.MinValue),
        "LastModifiedAt" => desc
            ? source.OrderByDescending(t => t.LastModifiedAt)
            : source.OrderBy(t => t.LastModifiedAt),
        _ => desc
            ? source.OrderByDescending(t => t.CreatedAt)
            : source.OrderBy(t => t.CreatedAt),
    };
}

private static IOrderedEnumerable<TaskItem> ThenByField(
    IOrderedEnumerable<TaskItem> source, string field, bool desc)
{
    return field switch
    {
        "CompletedAt" => desc
            ? source.ThenByDescending(t => t.CompletedAt ?? DateTime.MinValue)
            : source.ThenBy(t => t.CompletedAt ?? DateTime.MinValue),
        "LastModifiedAt" => desc
            ? source.ThenByDescending(t => t.LastModifiedAt)
            : source.ThenBy(t => t.LastModifiedAt),
        _ => desc
            ? source.ThenByDescending(t => t.CreatedAt)
            : source.ThenBy(t => t.CreatedAt),
    };
}
```

```csharp
/// <summary>
/// 整数比较转换器 — 用于 RadioButton 绑定整数值。
/// 当绑定的 int 值 == ConverterParameter 时返回 true。
/// 使用 Mode=TwoWay：选中时回写 ConverterParameter 代表的 int 值。
/// </summary>
public class IntToBoolConverter : System.Windows.Data.IValueConverter
{
    public object Convert(object value, Type targetType, object parameter, System.Globalization.CultureInfo culture)
    {
        if (value is int intVal && parameter is string paramStr && int.TryParse(paramStr, out int target))
            return intVal == target;
        return false;
    }

    public object ConvertBack(object value, Type targetType, object parameter, System.Globalization.CultureInfo culture)
    {
        if (value is bool isChecked && isChecked && parameter is string paramStr && int.TryParse(paramStr, out int target))
            return target;
        return System.Windows.Data.Binding.DoNothing;
    }
}
```

```csharp
using System.Collections.ObjectModel;
using System.Text;
using System.Windows;
using System.Windows.Input;
using Taskato.Models;
using Taskato.Services;
using Taskato.Utils;

namespace Taskato.ViewModels
{
    /// <summary>
    /// 每日总结窗体的 ViewModel
    /// 支持日期切换、统计数据计算、复制/导出
    /// </summary>
    public class DailySummaryViewModel : ViewModelBase
    {
        private readonly DatabaseService _dbService;
        private List<TaskItem> _allTasks = new();

        /// <summary>当前查看的日期</summary>
        private DateTime _currentDate = DateTime.Today;
        public DateTime CurrentDate
        {
            get => _currentDate;
            set
            {
                if (SetProperty(ref _currentDate, value))
                {
                    OnPropertyChanged(nameof(DateDisplay));
                    _ = LoadSummaryAsync();
                }
            }
        }

        /// <summary>日期显示文本（如 "2026-05-29 每日总结"）</summary>
        public string DateDisplay => $"{CurrentDate:yyyy-MM-dd} 每日总结";

        /// <summary>导语（顶部一句话）</summary>
        private string _summaryLine = string.Empty;
        public string SummaryLine
        {
            get => _summaryLine;
            set => SetProperty(ref _summaryLine, value);
        }

        /// <summary>总任务数</summary>
        private int _totalCount;
        public int TotalCount
        {
            get => _totalCount;
            set => SetProperty(ref _totalCount, value);
        }

        /// <summary>已完成数</summary>
        private int _completedCount;
        public int CompletedCount
        {
            get => _completedCount;
            set => SetProperty(ref _completedCount, value);
        }

        /// <summary>完成率文本</summary>
        private string _completionRate = string.Empty;
        public string CompletionRate
        {
            get => _completionRate;
            set => SetProperty(ref _completionRate, value);
        }

        /// <summary>高优先级完成数</summary>
        private int _highPriorityCompleted;
        public int HighPriorityCompleted
        {
            get => _highPriorityCompleted;
            set => SetProperty(ref _highPriorityCompleted, value);
        }

        /// <summary>已完成任务列表</summary>
        public ObservableCollection<TaskItem> CompletedTasks { get; } = new();

        /// <summary>未完成任务列表</summary>
        public ObservableCollection<TaskItem> UncompletedTasks { get; } = new();

        /// <summary>是否可以切换到前一天</summary>
        public bool CanGoPrev => CurrentDate > DateTime.Today.AddDays(-365);

        /// <summary>是否可以切换到后一天</summary>
        public bool CanGoNext => CurrentDate < DateTime.Today;

        public ICommand GoPrevCommand { get; }
        public ICommand GoNextCommand { get; }
        public ICommand CopyCommand { get; }
        public ICommand ExportCommand { get; }

        public DailySummaryViewModel(DatabaseService dbService)
        {
            _dbService = dbService;

            GoPrevCommand = new RelayCommand(_ =>
            {
                CurrentDate = CurrentDate.AddDays(-1);
                OnPropertyChanged(nameof(CanGoPrev));
                OnPropertyChanged(nameof(CanGoNext));
            });

            GoNextCommand = new RelayCommand(_ =>
            {
                if (CurrentDate < DateTime.Today)
                {
                    CurrentDate = CurrentDate.AddDays(1);
                    OnPropertyChanged(nameof(CanGoPrev));
                    OnPropertyChanged(nameof(CanGoNext));
                }
            });

            CopyCommand = new RelayCommand(_ =>
            {
                Clipboard.SetText(BuildSummaryText());
            });

            ExportCommand = new RelayCommand(_ =>
            {
                var saveDialog = new Microsoft.Win32.SaveFileDialog
                {
                    Filter = "文本文件|*.txt",
                    DefaultExt = ".txt",
                    FileName = $"Taskato_每日总结_{CurrentDate:yyyyMMdd}"
                };
                if (saveDialog.ShowDialog() == true)
                {
                    System.IO.File.WriteAllText(saveDialog.FileName, BuildSummaryText(),
                        new UTF8Encoding(true));
                }
            });
        }

        /// <summary>加载指定日期的总结数据</summary>
        public async Task LoadSummaryAsync()
        {
            var dayStart = CurrentDate.Date;
            var dayEnd = dayStart.AddDays(1);

            // 查询该日期创建的所有任务
            _allTasks = await _dbService.SearchTasksAsync(dayStart, dayStart);

            var completed = _allTasks.Where(t => t.IsCompleted).ToList();
            var uncompleted = _allTasks.Where(t => !t.IsCompleted).ToList();

            TotalCount = _allTasks.Count;
            CompletedCount = completed.Count;
            CompletionRate = TotalCount > 0 ? $"{(double)CompletedCount / TotalCount:P0}" : "N/A";
            HighPriorityCompleted = completed.Count(t => t.Priority >= 2);

            // 导语
            if (TotalCount == 0)
                SummaryLine = "今天没有任务记录。";
            else if (CompletedCount == TotalCount)
                SummaryLine = $"今天完成了全部 {TotalCount} 个任务，太棒了！";
            else
                SummaryLine = $"今天完成了 {CompletedCount}/{TotalCount} 个任务，再接再厉！";

            // 清单（按优先级降序）
            CompletedTasks.Clear();
            foreach (var t in completed.OrderByDescending(t => t.Priority))
                CompletedTasks.Add(t);

            UncompletedTasks.Clear();
            foreach (var t in uncompleted.OrderByDescending(t => t.Priority))
                UncompletedTasks.Add(t);
        }

        /// <summary>构建纯文本总结（用于复制和导出）</summary>
        private string BuildSummaryText()
        {
            var sb = new StringBuilder();
            sb.AppendLine($"{CurrentDate:yyyy-MM-dd} 每日总结");
            sb.AppendLine(new string('─', 30));
            sb.AppendLine();
            sb.AppendLine(SummaryLine);
            sb.AppendLine();
            sb.AppendLine($"统计：总计 {TotalCount} | 完成 {CompletedCount}（{CompletionRate}）| 高优先级完成 {HighPriorityCompleted}");
            sb.AppendLine();
            sb.AppendLine("✅ 已完成：");
            foreach (var t in CompletedTasks)
            {
                var pLabel = t.Priority switch { 3 => "[紧急]", 2 => "[高]", 1 => "[中]", _ => "[无]" };
                sb.AppendLine($"  {pLabel} {t.Title}");
            }
            sb.AppendLine();
            sb.AppendLine("❌ 未完成：");
            foreach (var t in UncompletedTasks)
            {
                var pLabel = t.Priority switch { 3 => "[紧急]", 2 => "[高]", 1 => "[中]", _ => "[无]" };
                sb.AppendLine($"  {pLabel} {t.Title}");
            }
            return sb.ToString();
        }
    }
}
```

```csharp
/// <summary>
/// 优先级整数转标签文本（3→"[紧急]", 2→"[高]", 1→"[中]", 0→"[无]"）
/// </summary>
public class PriorityToLabelConverter : System.Windows.Data.IValueConverter
{
    public object Convert(object value, Type targetType, object parameter, System.Globalization.CultureInfo culture)
    {
        if (value is int p)
        {
            return p switch
            {
                3 => "[紧急]",
                2 => "[高]",
                1 => "[中]",
                _ => "[无]"
            };
        }
        return "[无]";
    }

    public object ConvertBack(object value, Type targetType, object parameter, System.Globalization.CultureInfo culture)
        => throw new NotImplementedException();
}
```
